REGISTER
POST /auth/REGISTER
BODY firstName,lastName, emailOrMobile, password, confirmPassword
RESPONSE
201 { accessToken : String }
400 { message : String }
500 { message : String }

login
POST /auth/login
BODY emailOrMobile, password
RESPONSE
200 { accessToken : String, user: {id, firstName, lastName, profileImage} }
400 { message : String }
500 { message : String }

GET AUTHENTICATED USER 
GET /auth/me
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
RESPONSE
200 { user: {id, firstName, lastName, profileImage} }
401 { message : String }
500 { message : String }

UPDATE PROFILE/COVER profileImage
PATH /user
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
BODY profileImage?, coverImage?
RESPONSE
200 { profileImage? : String, coverImage? : String }
400 { message : String}
401 { message : String}
500 { message : string}

GET USER PROFILE BY ID
GET /user/:userId
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
RESPONSE
200 { user : { id : Number, firstName : String, ... } }
401 { message : String}
500 { message : string}
